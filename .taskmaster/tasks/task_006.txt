     const connect = async () => {

         const token = md5(password + salt);
         
         // Test connection
         const response = await fetch(
           `${server}/rest/ping.view?v=1.13.0&c=deplayer&f=json&u=${username}&s=${salt}&t=${token}`
         );
         
         const data = await response.json();
         if (data['subsonic-response'].status === 'ok') {
           setIsConnected(true);
           // Save connection details to settings
           return true;
         }
         return false;
       } catch (error) {
         console.error('Subsonic connection error:', error);
         return false;
       }
     };
     
     const fetchLibrary = async () => {
       // Implement fetching library from Subsonic server
     };

       const salt = generateRandomString(10);
       const token = md5(password + salt);
       return `${server}/rest/stream.view?id=${id}&v=1.13.0&c=deplayer&f=json&u=${username}&s=${salt}&t=${token}`;
