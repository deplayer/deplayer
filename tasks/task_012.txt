   ```typescript
   import { ElectricDatabase, electrify } from 'electric-sql/wa-sqlite';
   
   async function setupSync(dbName: string, syncServerUrl: string) {
     // Initialize the database
     const conn = await ElectricDatabase.init(dbName);
     
     // Define the shape of your database
     const schema = {
       version: 1,
       tables: {
         media: {
           columns: {
             id: { type: 'TEXT', primaryKey: true },
             title: { type: 'TEXT' },
             artist: { type: 'TEXT' },
             album: { type: 'TEXT' },
             duration: { type: 'INTEGER' },
             path: { type: 'TEXT' },
             type: { type: 'TEXT' },
             provider: { type: 'TEXT' },
             thumbnail: { type: 'TEXT' },
             added_date: { type: 'INTEGER' }
           }
         },
         playlists: {
           columns: {
             id: { type: 'TEXT', primaryKey: true },
             name: { type: 'TEXT' },
             description: { type: 'TEXT' },
             created_date: { type: 'INTEGER' }
           }
         },
         playlist_items: {
           columns: {
             playlist_id: { type: 'TEXT' },
             media_id: { type: 'TEXT' },
             position: { type: 'INTEGER' }
           },
           primaryKey: ['playlist_id', 'media_id']
         },
         settings: {
           columns: {
             key: { type: 'TEXT', primaryKey: true },
             value: { type: 'TEXT' }
           }
         }
       }
     };
     
     // Electrify the database
     const electric = await electrify(conn, schema);
     
     // Connect to the sync server
     await electric.connect(syncServerUrl, {
       auth: { token: 'auth-token' } // Implement proper auth
     });
     
     return electric;
   }
   ```
