1. Create a YoutubeDLProvider component:
   ```typescript
   function useYoutubeDLProvider() {
     const [serverUrl, setServerUrl] = useState('');
     const [isConnected, setIsConnected] = useState(false);
     
     const testConnection = async () => {
       try {
         const response = await fetch(`${serverUrl}/api/version`);
         const data = await response.json();
         if (data.version) {
           setIsConnected(true);
           return true;
         }
         return false;
       } catch (error) {
         console.error('youtube-dl-server connection error:', error);
         return false;
       }
     };
     
     const fetchVideoInfo = async (url: string) => {
       try {
         const response = await fetch(`${serverUrl}/api/info?url=${encodeURIComponent(url)}`);
         return await response.json();
       } catch (error) {
         console.error('Error fetching video info:', error);
         return null;
       }
     };
     
     const downloadVideo = async (url: string, format: string = 'best') => {
       try {
         const response = await fetch(`${serverUrl}/api/download`, {
           method: 'POST',
           headers: {
             'Content-Type': 'application/json'
           },
           body: JSON.stringify({ url, format })
         });
         return await response.json();
       } catch (error) {
         console.error('Error downloading video:', error);
         return null;
       }
     };
     
     return {
       testConnection,
       fetchVideoInfo,
       downloadVideo,
       setServerUrl,
       isConnected
     };
   }
   ```
