   function useTorrentProvider() {
     const [client, setClient] = useState<WebTorrent.Instance | null>(null);
     const [torrents, setTorrents] = useState<WebTorrent.Torrent[]>([]);
     
     useEffect(() => {
       const webTorrentClient = new WebTorrent();
       setClient(webTorrentClient);
       
       return () => {
         webTorrentClient.destroy();
       };
     }, []);
     
     const addTorrent = useCallback((torrentId: string) => {
       if (!client) return;
       
       client.add(torrentId, (torrent) => {
         // Process torrent files and add to library
         const mediaFiles = torrent.files.filter(file => {
           const ext = file.name.split('.').pop()?.toLowerCase();
           return ['mp3', 'mp4', 'wav', 'ogg', 'webm'].includes(ext || '');
         });
         
         // Add media files to database
       });
     }, [client]);
     
     return { addTorrent, torrents };
   }
